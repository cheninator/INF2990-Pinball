89 void NoeudComposite::vider()
90 {
91  // L'iteration doit etre faite ainsi pour eviter les problemes lorsque
92  // le desctructeur d'un noeud modifie l'arbre, par exemple en retirant
93  // d'autres noeuds. Il pourrait y avoir une boucle infinie si la
94  // desctruction d'un enfant entraînerait l'ajout d'un autre.
95  while (!enfants_.empty()) {
96  NoeudAbstrait* enfantAEffacer{ enfants_.front() };
97  enfants_.erase(enfants_.begin());
98  delete enfantAEffacer;
99  }
100 }

296 void NoeudComposite::effacerSelection()
297 {
298  // On efface tous les noeuds selectionnes descendants des enfants.
299  for (NoeudAbstrait * enfant : enfants_){
300  enfant->effacerSelection();
301  }
302 
303  // On efface les enfants selectionnes. On effectue ce traitement
304  // dans une seconde boucle pour eviter de faire des assomptions
305  // sur la robustesse des iterateurs lorsque le conteneur est
306  // modifie pendant une iteration.
307  for (conteneur_enfants::iterator it{ enfants_.begin() };
308  it != enfants_.end();
309  ) {
310  if ((*it)->estSelectionne()) {
311  NoeudAbstrait* enfant{ (*it) };
312  enfants_.erase(it);
313  delete enfant;
314 
315  // On ramene l'iteration au debut de la boucle, car le destructeur
316  // de l'enfant pourrait eventuellement avoir retire d'autres

318  // eventuellement etre evite avec des iterateurs plus robustes.

321  }
