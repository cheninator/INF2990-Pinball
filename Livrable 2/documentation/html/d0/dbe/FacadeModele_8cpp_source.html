538 void FacadeModele::tournerSelectionSouris(int x1, int y1, int x2, int y2)
539 {
540  // Visiter l'arbre pour trouver le centre de masse des noeuds selectionnes
541  VisiteurCentreDeMasse visCM;
542  arbre_->accepterVisiteur(&visCM);
543  glm::dvec3 centreRotation = visCM.obtenirCentreDeMasse();
544 
545  // On calcule l'angle de la rotation:
546  double angle = (y2 - y1) / 3.0;
547  double angleEnRadian = angle * utilitaire::PI_180;
548  glm::dmat3 transform = glm::dmat3{ glm::dvec3{ cos(-angleEnRadian), -sin(-angleEnRadian), 0 },
549  glm::dvec3{ sin(-angleEnRadian), cos(-angleEnRadian), 0 },
550  glm::dvec3{ 0, 0, 1 } };
551 
552  // On decide si la rotation peut se faire
553  bool onTourne = true;
554 
555  // Obtenir une liste des points englobants des noeud selectionnes
556  VisiteurListeEnglobante visLE;
557  arbre_->accepterVisiteur(&visLE);
558 
559  for (conteneur_boite_englobante boite : visLE.obtenirListeEnglobante())
560  {
561  std::vector<glm::dvec3> pointsTransformes;
562  glm::dvec3 nouvellePosition = centreRotation + transform * (boite.second->obtenirPositionRelative() - centreRotation);
563  if (boite.first.size() == 1)
564  {
565  double rayon = boite.first[0].x;
566 
567  pointsTransformes.push_back(nouvellePosition + glm::dvec3{ rayon, 0, 0 });
568  pointsTransformes.push_back(nouvellePosition + glm::dvec3{ -rayon, 0, 0 });
569  pointsTransformes.push_back(nouvellePosition + glm::dvec3{ 0, rayon, 0 });
570  pointsTransformes.push_back(nouvellePosition + glm::dvec3{ 0, -rayon, 0 });
571  }
572  else
573  {
574  for (glm::dvec3 vecteur : boite.first)
575  {
576  pointsTransformes.push_back(nouvellePosition + transform * vecteur);
577  }
578  }
579  for (glm::dvec3 point : pointsTransformes)
580  {
581  if (!estDansTable(point))
582  return;
583  }
584  }
585 
586  // Visiter l'arbre et faire la rotation.
587  if (onTourne)
588  {
589  VisiteurRotationPoint visSP(glm::dvec3{ 0, 0, angle }, centreRotation);
590  arbre_->accepterVisiteur(&visSP);
591  }

613 void FacadeModele::agrandirSelection(int x1, int y1, int x2, int y2)
614 {
615  double scale = glm::exp((y1 - y2) * glm::log(1.003)); // exp(b log(a)) = a^b
616  // Pour agrandir on multiplie le scale courrant par 1.003 et ce une fois pour chaque deplacement en y
617  // donc on multiplie par 1.003^(y1-y2).
618  // Si (y1-y2) est negatif, ceci va nous faire diviser par 1.003, donc l'objet va rapetisser.
619 
620  // Au final, on multiplie le scale courrant par 1.003 une fois pour chaque deplacement elementaire vers le haut,
621  // On divise par 1.003 pour chaque deplacement elementaire vers le bas.
622  double distanceConstante = 1.0;
623  double correctionZ = (1 - scale) * distanceConstante;
624 
625  VisiteurListeEnglobante visLE;
626  arbre_->accepterVisiteur(&visLE);
627 
628  for (conteneur_boite_englobante boite : visLE.obtenirListeEnglobante())
629  {
630  std::vector<glm::dvec3> pointsTransformes;
631  glm::dvec3 position = boite.second->obtenirPositionRelative();
632  if (boite.first.size() == 1)
633  {
634  double rayon = boite.first[0].x;
635 
636  pointsTransformes.push_back(position + scale *glm::dvec3{ rayon, 0, 0 });
637  pointsTransformes.push_back(position + scale *glm::dvec3{ -rayon, 0, 0 });
638  pointsTransformes.push_back(position + scale *glm::dvec3{ 0, rayon, 0 });
639  pointsTransformes.push_back(position + scale *glm::dvec3{ 0, -rayon, 0 });
640  }
641  else
642  {
643  for (glm::dvec3 vecteur : boite.first)
644  {
645  pointsTransformes.push_back(position + scale * vecteur);
646  }
647  }
648  for (glm::dvec3 point : pointsTransformes)
649  {
650  if (!estDansTable(point))
651  return;
652  }
653  }
654 
655  VisiteurAgrandissement visAgr(glm::dvec3{ scale, scale, scale });
656  arbre_->accepterVisiteur(&visAgr);
657 
658  VisiteurDeplacement visDep(glm::dvec3{ 0, 0, correctionZ });
659  arbre_->accepterVisiteur(&visDep);
660 
661 }
