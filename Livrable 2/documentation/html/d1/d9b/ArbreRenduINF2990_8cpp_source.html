421  BYTE* pixels = new BYTE[3 * FACTEUR_DEBUG * width * height];
422  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
423

426 
427  // Convert to FreeImage format & save to file
428  FIBITMAP* image = FreeImage_ConvertFromRawBits(pixels, width, height, ((((24 * width) + 31) / 32) * 4), 24,
429  0x000000, 0x000000, 0x000000, false);
430  // La couleur ici semble... useless (le 0x000000)
431

433  if (maxSize != 0)

435
