174  if (points.size() > 1)

178

180  for (unsigned int i = 0; i < points.size(); i++)

183  return false;
184  }
185

189

191  else if (points.size() == 1)

194  double rayon = points[0].x;

197  glm::dvec3 sud = { position.x, position.y - rayon, position.z };
