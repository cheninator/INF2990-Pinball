298

300  {

302  vider();
303 
304  // Tenter d'obtenir l'element table, puis l'attribut nbEnfants

306 
307  if (elementTable != nullptr)

309  // Creer la table et l'ajouter a l'arbre de rendu

315

318 
319  const char* positionX = "posX";

321  const char* positionZ = "posZ";

326  const char* angleY = "angleY";
327  const char* angleZ = "angleZ";
328 
329  // Creer les enfants et les ajouter a la table

350  noeudConcret->setColorShift(enfant->FindAttribute("color")->BoolValue());
351 
352  // Ajouter l'enfant a la table
353  table->ajouter(noeudConcret);
354 
355  // Le cas ou le noeud cree est un portail
356  if (noeudConcret->obtenirType() == "portail")

358  // Nombre actuel d'enfants de la table

371  // Traiter le frere de droite de l'enfant
372  enfant = enfant->NextSiblingElement();
373  }
374
